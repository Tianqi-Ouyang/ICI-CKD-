3.1 standardize_colnames

standardize_colnames <- function(df) {
  names(df) <- gsub("[^[:alnum:]_]", "_", names(df))
  names(df) <- gsub("_+", "_", names(df))
  names(df) <- gsub("_$", "", names(df))
  return(df)
}

3.2 ckd_epi_gfr (race-free 2021 CKD-EPI)

ckd_epi_gfr <- function(creat, male, age) {
  k <- ifelse(male == 1, 0.9, 0.7)
  a <- ifelse(male == 1, -0.302, -0.241)
  m <- ifelse(male == 1, 1, 1.012)
  gfr <- 142 * min(c(creat/k, 1))^a *
               max(c(creat/k, 1))^-1.2 *
               0.9938^age * m
  return(gfr)
}
gfr_calc <- Vectorize(ckd_epi_gfr)

3.3 get_ckd_stage

get_ckd_stage <- function(egfr) {
  if (is.na(egfr))      return(NA)
  else if (egfr >= 90)  return(1)
  else if (egfr >= 60)  return(2)
  else if (egfr >= 30)  return(3)
  else if (egfr >= 15)  return(4)
  else                  return(5)
}

3.4 pre_lab — closest lab on or before index date

pre_lab <- function(data, month, time, index_date, Group_Id) {
  index_date <- enquo(index_date)
  time_value <- if (month) time * 30 else time

  lab_name      <- paste0("pre_", Group_Id, "_", time_value, "days")
  lab_date_name <- paste0("pre_", Group_Id, "_", time_value, "days_date")

  data %>%
    filter(Group_Id == !!Group_Id, grepl("^\\d+\\.\\d+$", Result)) %>%
    filter(!is.na(Result)) %>%
    mutate(Result = as.numeric(Result)) %>%
    filter(Seq_Date_Time >= (!!index_date) - time_value,
           Seq_Date_Time <=  !!index_date) %>%
    arrange(EMPI, desc(Seq_Date_Time)) %>%
    distinct(EMPI, .keep_all = TRUE) %>%
    rename(!!lab_name      := Result,
           !!lab_date_name := Seq_Date_Time) %>%
    select(EMPI, !!lab_name, !!lab_date_name)
}

3.5 med_list — patients on a medication on/before index date

FIXED 2026-04-17 — Temporal leakage bug (flagged by Codex adversarial review). Previous version only applied Medication_Date <= index_date when excluded_names was supplied; the else branch dropped the date guard, so exposures like ppi, ace_arb, and diu could include medications started AFTER ICI. This contaminated “baseline” covariates with post-ICI treatment history and biased downstream models. Date filter is now applied in both branches. Any final_master_*.xlsx built from the old version must be regenerated before use.

med_list <- function(names, index_date, excluded_names = NULL, data) {
  # Always enforce the pre-index date guard — do not remove.
  filtered <- data %>%
    filter(Medication_Date <= {{index_date}})

  if (!is.null(excluded_names)) {
    filtered %>%
      filter(grepl(paste(names, collapse = "|"), Medication, ignore.case = TRUE) &
            !grepl(paste(excluded_names, collapse = "|"), Medication, ignore.case = TRUE)) %>%
      distinct(EMPI)
  } else {
    filtered %>%
      filter(grepl(paste(names, collapse = "|"), Medication, ignore.case = TRUE)) %>%
      distinct(EMPI)
  }
}

3.6 dia_list_icd / dia_list_icd_after_index / dia_list_name

dia_list_icd <- function(data, index_date, icd_code,
                         excluded_names = NULL, med_data = NULL, med_names = NULL) {
  if (is.null(excluded_names)) {
    dia_result <- data %>%
      filter(Date <= {{index_date}}) %>%
      filter(grepl(paste(icd_code, collapse = "|"), Code, ignore.case = TRUE)) %>%
      distinct(EMPI)
  } else {
    dia_result <- data %>%
      filter(Date <= {{index_date}}) %>%
      filter(grepl(paste(icd_code,      collapse = "|"), Code,            ignore.case = TRUE),
            !grepl(paste(excluded_names, collapse = "|"), Diagnosis_Name,  ignore.case = TRUE)) %>%
      distinct(EMPI)
  }

  if (!is.null(med_data)) {
    med_result <- med_data %>%
      filter(Medication_Date <= {{index_date}}) %>%
      filter(grepl(paste(med_names, collapse = "|"), Medication, ignore.case = TRUE)) %>%
      distinct(EMPI)
  } else {
    med_result <- NULL
  }

  rbind(dia_result, med_result) %>% distinct(EMPI)
}
dia_list_icd_after_index <- function(data, index_date, icd_code,
                                     excluded_names = NULL,
                                     med_data = NULL, med_names = NULL) {
  dia_subset <- data %>%
    filter(Date > {{index_date}}) %>%
    filter(grepl(paste(icd_code, collapse = "|"), Code, ignore.case = TRUE))

  if (!is.null(excluded_names)) {
    dia_subset <- dia_subset %>%
      filter(!grepl(paste(excluded_names, collapse = "|"), Diagnosis_Name, ignore.case = TRUE))
  }

  dia_result <- dia_subset %>%
    group_by(EMPI) %>%
    summarise(event_date = min(Date), .groups = "drop")

  med_result <- NULL
  if (!is.null(med_data) && !is.null(med_names)) {
    med_result <- med_data %>%
      filter(Medication_Date > {{index_date}}) %>%
      filter(grepl(paste(med_names, collapse = "|"), Medication, ignore.case = TRUE)) %>%
      group_by(EMPI) %>%
      summarise(event_date = min(Medication_Date), .groups = "drop")
  }

  bind_rows(dia_result, med_result) %>%
    group_by(EMPI) %>%
    summarise(first_event_date = min(event_date), .groups = "drop")
}

3.7 make_ckd_outcome_date — sustained-episode outcomes

Identifies the first episode where a condition (e.g., eGFR < 60) is sustained for ≥ days_threshold days and returns both the binary flag and first-event date.

make_ckd_outcome_date <- function(data, id_var, date_var, condition_expr,
                                  days_threshold = 90, outcome_name = "default") {
  id_var      <- enquo(id_var)
  date_var    <- enquo(date_var)
  cond_q      <- enquo(condition_expr)
  outcome_sym <- ensym(outcome_name)
  outcome_date_name <- paste0(outcome_name, "_date")

  working <- data %>%
    arrange(!!id_var, !!date_var) %>%
    group_by(!!id_var) %>%
    mutate(
      .cond   = if_else(!!cond_q, 1L, 0L),
      .change = .cond != lag(.cond, default = first(.cond)),
      .run    = cumsum(.change)
    )

  episodes <- working %>%
    group_by(!!id_var, .run) %>%
    summarise(.code = first(.cond),
              .start = first(!!date_var),
              .last  = last(!!date_var),
              .groups = "drop") %>%
    group_by(!!id_var) %>%
    mutate(.end  = lead(.start),
           .end  = if_else(is.na(.end), .last, .end),
           .days = as.integer(.end - .start)) %>%
    ungroup()

  episodes %>%
    group_by(!!id_var) %>%
    summarise(
      has_event  = any(.code == 1L & .days >= days_threshold),
      event_date = if (any(.code == 1L & .days >= days_threshold))
                      min(.start[.code == 1L & .days >= days_threshold])
                   else as.Date(NA),
      .groups = "drop"
    ) %>%
    mutate(!!outcome_sym := as.integer(has_event)) %>%
    rename(!!outcome_date_name := event_date) %>%
    select(-has_event)
}

3.8 create_aki — KDIGO-style AKI from serial CRE

create_aki <- function(data,
                       index_date = ICI_Dose_Date,
                       Group_Id   = EMPI,
                       median_window_days = 90,
                       lowest_window_days = 2) {
  index_date <- enquo(index_date)
  Group_Id   <- enquo(Group_Id)

  df <- data %>%
    filter(Group_Id == "CRE", grepl("^\\d+\\.\\d+$", Result)) %>%
    filter(!is.na(Result)) %>%
    mutate(Result = as.numeric(Result)) %>%
    rename(cre_date = Seq_Date_Time, cre = Result) %>%
    select(EMPI, cre, cre_date, !!index_date) %>%
    arrange(!!Group_Id, cre_date)

  df %>%
    filter(cre_date > !!index_date) %>%
    group_by(!!Group_Id) %>%
    mutate(
      median_cre_90days = sapply(seq_along(cre), function(i) {
        vals <- cre[cre_date >= cre_date[i] - median_window_days &
                    cre_date <= cre_date[i] - 1]
        if (length(vals) == 0) NA_real_ else median(vals, na.rm = TRUE)
      }),
      lowest_cre_2days = sapply(seq_along(cre), function(i) {
        vals <- cre[cre_date >= cre_date[i] - lowest_window_days &
                    cre_date <= cre_date[i] - 1]
        if (length(vals) == 0) NA_real_ else min(vals, na.rm = TRUE)
      }),
      aki_flag = (!is.na(median_cre_90days) & cre > 1.5 * median_cre_90days) |
                 (!is.na(lowest_cre_2days)  & cre - lowest_cre_2days >= 0.3)
    ) %>%
    filter(aki_flag) %>%
    arrange(!!Group_Id, cre_date) %>%
    summarise(aki      = 1L,
              aki_date = min(cre_date), .groups = "drop")
}

left_join_all <- function(master, df_list, by) {

ckd_epi_gfr <- function(creat, male, age) {
  k <- ifelse(male == 1, 0.9, 0.7)
  a <- ifelse(male == 1, -0.302, -0.241)
  m <- ifelse(male == 1, 1, 1.012)
  gfr <- 142 * min(c(creat/k, 1))^a *
               max(c(creat/k, 1))^-1.2 *
               0.9938^age * m
  return(gfr)
}

## 3.7 make_ckd_outcome_date — sustained-episode outcomes

Identifies the first episode where a condition (e.g., `eGFR < 60`) is sustained
for ≥ `days_threshold` days and returns both the binary flag and first-event date.

```{r}
#| label: fn-make-ckd-outcome-date
make_ckd_outcome_date <- function(data, id_var, date_var, condition_expr,
                                  days_threshold = 90, outcome_name = "default") {
  id_var      <- enquo(id_var)
  date_var    <- enquo(date_var)
  cond_q      <- enquo(condition_expr)
  outcome_sym <- ensym(outcome_name)
  outcome_date_name <- paste0(outcome_name, "_date")

  working <- data %>%
    arrange(!!id_var, !!date_var) %>%
    group_by(!!id_var) %>%
    mutate(
      .cond   = if_else(!!cond_q, 1L, 0L),
      .change = .cond != lag(.cond, default = first(.cond)),
      .run    = cumsum(.change)
    )

  episodes <- working %>%
    group_by(!!id_var, .run) %>%
    summarise(.code = first(.cond),
              .start = first(!!date_var),
              .last  = last(!!date_var),
              .groups = "drop") %>%
    group_by(!!id_var) %>%
    mutate(.end  = lead(.start),
           .end  = if_else(is.na(.end), .last, .end),
           .days = as.integer(.end - .start)) %>%
    ungroup()

  episodes %>%
    group_by(!!id_var) %>%
    summarise(
      has_event  = any(.code == 1L & .days >= days_threshold),
      event_date = if (any(.code == 1L & .days >= days_threshold))
                      min(.start[.code == 1L & .days >= days_threshold])
                   else as.Date(NA),
      .groups = "drop"
    ) %>%
    mutate(!!outcome_sym := as.integer(has_event)) %>%
    rename(!!outcome_date_name := event_date) %>%
    select(-has_event)
}
```

## 3.8 create_aki — KDIGO-style AKI from serial CRE

```{r}
#| label: fn-create-aki
create_aki <- function(data,
                       index_date = ICI_Dose_Date,
                       Group_Id   = EMPI,
                       median_window_days = 90,
                       lowest_window_days = 2) {
  index_date <- enquo(index_date)
  Group_Id   <- enquo(Group_Id)

  df <- data %>%
    filter(Group_Id == "CRE", grepl("^\\d+\\.\\d+$", Result)) %>%
    filter(!is.na(Result)) %>%
    mutate(Result = as.numeric(Result)) %>%
    rename(cre_date = Seq_Date_Time, cre = Result) %>%
    select(EMPI, cre, cre_date, !!index_date) %>%
    arrange(!!Group_Id, cre_date)

  df %>%
    filter(cre_date > !!index_date) %>%
    group_by(!!Group_Id) %>%
    mutate(
      median_cre_90days = sapply(seq_along(cre), function(i) {
        vals <- cre[cre_date >= cre_date[i] - median_window_days &
                    cre_date <= cre_date[i] - 1]
        if (length(vals) == 0) NA_real_ else median(vals, na.rm = TRUE)
      }),
      lowest_cre_2days = sapply(seq_along(cre), function(i) {
        vals <- cre[cre_date >= cre_date[i] - lowest_window_days &
                    cre_date <= cre_date[i] - 1]
        if (length(vals) == 0) NA_real_ else min(vals, na.rm = TRUE)
      }),
      aki_flag = (!is.na(median_cre_90days) & cre > 1.5 * median_cre_90days) |
                 (!is.na(lowest_cre_2days)  & cre - lowest_cre_2days >= 0.3)
    ) %>%
    filter(aki_flag) %>%
    arrange(!!Group_Id, cre_date) %>%
    summarise(aki      = 1L,
              aki_date = min(cre_date), .groups = "drop")
}
```
